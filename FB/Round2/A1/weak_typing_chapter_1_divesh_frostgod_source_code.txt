const int inf = (ll)1e18 + 5ll;
const int mod =   (int)998244353;

int power(int a, int b){
    int ans =1ll;
    while(b > 0){
        if(b%2ll == 1ll){
            ans = (ans * a)%mod;
        }

        a = (a * a)%mod;
        b /= 2ll;
    }
    return ans;
}

int gcd(int a, int b)
{
    if (b == 0)
        return a;
    return gcd(b, a % b);
     
}

void solve(){
    int n;
    cin >> n;
    string s;
    cin >> s;

    int ans =0;
    for(int i=1; i<n; i++){

        if(s[i] == 'F'){
            s[i] = s[i-1];
            continue;
        }

        else if(s[i-1] == 'F'){
            continue;
        }

        if(s[i] != s[i-1]){
            ans += 1;
        }
    }
    cout << ans << "\n";
}

    int cnt = 1;
    while(t--){
        cout << "Case #" << cnt << ":";
        solve();

    }
